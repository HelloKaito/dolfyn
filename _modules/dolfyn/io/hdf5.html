[docs]def is_pydicth5(fname):
    retval = False
    fd = h5.File(fname, mode='r')
    if fd.attrs.get('__package_name__', None) in [b'pyDictH5', 'pyDictH5']:
        retval = True
    fd.close()
    return retval
